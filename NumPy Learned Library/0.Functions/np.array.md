





```python
import numpy as np

# Boolean array
bool_array = np.array([True, False, True], dtype=bool)

# Integer array
int_array = np.array([1, 2, 3], dtype=np.int32)

# Floating-point array
float_array = np.array([1.0, 2.5, 3.7], dtype=np.float64)

# Complex number array
complex_array = np.array([1+2j, 3+4j], dtype=np.complex128)

# String array
string_array = np.array(['hello', 'world'], dtype=str)

# Object array (mixed types)
object_array = np.array([1, 'placeholder', 3.5], dtype=object)

# Datetime array
datetime_array = np.array(['2023-10-01', '2023-10-02'], dtype='datetime64')

# Timedelta array
timedelta_array = np.array(['1 days', '2 hours'], dtype='timedelta64')
```
